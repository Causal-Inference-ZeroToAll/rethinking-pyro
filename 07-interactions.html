def m7_1_model(rugged, log_gdp):
    a = pyro.sample("a", dist.Normal(8, 100))
    bR = pyro.sample("bR", dist.Normal(0, 1))
    mu = a + bR * rugged
    sigma = pyro.sample("sigma", dist.Uniform(0, 10))
    with pyro.plate("plate"):
        pyro.sample("log_gdp", dist.Normal(mu, sigma), obs=log_gdp)

pyro.set_rng_seed(5)
m7_1 = MAP("m7.1", m7_1_model, rugged=d_A1_rugged, log_gdp=d_A1_log_gdp)
m7_1.fit()

      def m7_5_model(rugged, cont_africa, log_gdp):
    a = pyro.sample("a", dist.Normal(8, 100))
    bR = pyro.sample("bR", dist.Normal(0, 1))
    bAR = pyro.sample("bAR", dist.Normal(0, 1))
    gamma = bR + bAR * cont_africa
    bA = pyro.sample("bA", dist.Normal(0, 1))
    mu = a + gamma * rugged + bA * cont_africa
    sigma = pyro.sample("sigma", dist.Uniform(0, 10))
    with pyro.plate("plate"):
        pyro.sample("log_gdp", dist.Normal(mu, sigma), obs=log_gdp)

pyro.set_rng_seed(3)
